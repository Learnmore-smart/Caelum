+        private async void DuplicateFile_Click(object sender, RoutedEventArgs e)
+        {
+            if (sender is FrameworkElement element && element.DataContext is HomeTile tile && !string.IsNullOrWhiteSpace(tile.Path))
+            {
+                if (!System.IO.File.Exists(tile.Path))
+                {
+                    await ShowDialogAsync("Error", "File not found.");
+                    await RemoveMissingRecentTileAsync(tile);
+                    return;

+
+                try
+                {
+                    string directory = System.IO.Path.GetDirectoryName(tile.Path);
+                    string fileNameWithoutExtension = System.IO.Path.GetFileNameWithoutExtension(tile.Path);
+                    string extension = System.IO.Path.GetExtension(tile.Path);
+                    string newPath = System.IO.Path.Combine(directory, $"{fileNameWithoutExtension} - Copy{extension}");
+                    int count = 1;
+                    while (System.IO.File.Exists(newPath))
+                    {
+                        count++;
+                        newPath = System.IO.Path.Combine(directory, $"{fileNameWithoutExtension} - Copy ({count}){extension}");
+                    }
+
+                    System.IO.File.Copy(tile.Path, newPath);
+                    await AddToRecentFilesAsync(newPath);
+                }
+                catch (Exception ex)
+                {
+                    await ShowDialogAsync("Error", $"Failed to duplicate file:\n{ex.Message}");
+                }
+            }

+        private async void RenameFile_Click(object sender, RoutedEventArgs e)
+        {
+            if (sender is FrameworkElement element && element.DataContext is HomeTile tile && !string.IsNullOrWhiteSpace(tile.Path))
+            {
+                if (!System.IO.File.Exists(tile.Path))
+                {
+                    await ShowDialogAsync("Error", "File not found.");
+                    await RemoveMissingRecentTileAsync(tile);
+                    return;
+                }
+
+                TextBox inputTextBox = new TextBox
+                {
+                    Text = System.IO.Path.GetFileNameWithoutExtension(tile.Path),
+                    AcceptsReturn = false
+                };
+                inputTextBox.Loaded += (s, ev) => inputTextBox.SelectAll();
+
+                ContentDialog renameDialog = new ContentDialog
+                {
+                    Title = "Rename file",
+                    Content = inputTextBox,
+                    PrimaryButtonText = "Rename",
+                    CloseButtonText = "Cancel",
+                    DefaultButton = ContentDialogButton.Primary,
+                    XamlRoot = this.XamlRoot
+                };
+
+                var result = await renameDialog.ShowAsync();
+                if (result == ContentDialogResult.Primary)
+                {
+                    string newName = inputTextBox.Text.Trim();
+                    if (string.IsNullOrWhiteSpace(newName))
+                    {
+                        return;
+                    }
+
+                    if (newName.IndexOfAny(System.IO.Path.GetInvalidFileNameChars()) >= 0)
+                    {
+                        await ShowDialogAsync("Error", "Invalid file name (cannot contain characters like / \\ : * ? \" < > |).");
+                        return;
+                    }
+
+                    string newPath = System.IO.Path.Combine(System.IO.Path.GetDirectoryName(tile.Path), newName + System.IO.Path.GetExtension(tile.Path));
+
+                    if (string.Equals(newPath, tile.Path, StringComparison.OrdinalIgnoreCase))
+                    {
+                        return;
+                    }
+
+                    if (System.IO.File.Exists(newPath))
+                    {
+                        await ShowDialogAsync("Error", "A file with that name already exists.");
+                        return;
+                    }
+
+                    try
+                    {
+                        System.IO.File.Move(tile.Path, newPath);
+
+                        // Replace the tile in collection
+                        int index = HomeTiles.IndexOf(tile);
+                        if (index >= 0)
+                        {
+                            HomeTiles[index] = HomeTile.CreateFileTile(newPath);
+                            await SaveRecentFilesAsync();
+                        }
+                    }
+                    catch (Exception ex)
+                    {
+                        await ShowDialogAsync("Error", $"Failed to rename file:\n{ex.Message}");
+                    }
+                }
+            }
+        }
